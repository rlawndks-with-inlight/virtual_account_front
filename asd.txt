server {
listen 80;
server_name emerypay.com
root html;


location / {
        return 301 https://emerypay.com$request_uri;
    }
}

server {

    listen       443 ssl;
    server_name  emerypay.com;

    client_max_body_size 256M;

    ssl_certificate     /etc/letsencrypt/live/emerypay.com/fullchain.pem;
    ssl_certificate_key /etc/letsencrypt/live/emerypay.com/privkey.pem;

    location / {
       proxy_pass http://emerypay.com:3000;
        proxy_set_header X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header X-Forwarded-Proto $scheme;
    }

    location = /50x.html {
        root   /usr/share/nginx/html;
    }
}
